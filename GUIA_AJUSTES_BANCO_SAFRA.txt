  ALTERACAO 2: MIGRAR DO CLOUDFLARE PARA SERVIDOR INTERNO
================================================================================

A aplicacao tem 2 partes:
  - Frontend: HTML/CSS/JS estatico (pasta "out/" gerada pelo Next.js)
  - Backend: 2 endpoints de API (chat e suggestions)

O frontend nao precisa de servidor especial - sao arquivos estaticos.
So o backend precisa ser migrado.

  npm install express cors dotenv
  npm uninstall wrangler


--- PASSO 2: Criar server.js na raiz do projeto ---

  const express = require('express');
  const cors = require('cors');
  const path = require('path');
  require('dotenv').config();

  const app = express();
  app.use(cors());
  app.use(express.json());
  app.use(express.static(path.join(__dirname, 'out')));

  // POST /api/chat (migrado de functions/api/chat.ts)
  app.post('/api/chat', async (req, res) => {
    const { spreadsheetText, messages } = req.body;

    const truncated = spreadsheetText.length > 12000
      ? spreadsheetText.slice(0, 12000) + '\n... (dados truncados)'
      : spreadsheetText;

    // Manter o mesmo systemPrompt que esta em chat.ts (linhas 133-145)
    const systemPrompt = `Voce e um assistente especialista em analise de dados do Banco Safra...`;

    const fullMessages = [
      { role: "user", content: systemPrompt },
      { role: "assistant", content: "Entendido. Pode perguntar." },
      ...messages,
    ];

    try {
      const response = await fetch(process.env.COPILOT_STUDIO_ENDPOINT, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${process.env.COPILOT_STUDIO_API_KEY}`,
        },
        body: JSON.stringify({
          type: "message",
          text: fullMessages[fullMessages.length - 1].content,
          channelData: { spreadsheetContext: truncated }
        }),
      });
      const data = await response.json();
      const reply = data.reply || data.text || data.activities?.[0]?.text;
      if (reply) return res.json({ reply });
      res.status(502).json({ error: 'Resposta vazia' });
    } catch (e) {
      res.status(500).json({ error: e.message });
    }
  });

  // POST /api/suggestions (migrado de functions/api/suggestions.ts)
  app.post('/api/suggestions', async (req, res) => {
    const { spreadsheetText } = req.body;

    const truncated = spreadsheetText.length > 8000
      ? spreadsheetText.slice(0, 8000) + '\n... (dados truncados)'
      : spreadsheetText;

    // Manter o mesmo prompt que esta em suggestions.ts (linhas 118-122)
    const prompt = `Analise esta planilha e gere exatamente 4 perguntas curtas...`;

    try {
      const response = await fetch(process.env.COPILOT_STUDIO_ENDPOINT, {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${process.env.COPILOT_STUDIO_API_KEY}`,
        },
        body: JSON.stringify({
          type: "message",
          text: prompt + '\n\n' + truncated,
        }),
      });
      const data = await response.json();
      const raw = data.reply || data.text || data.activities?.[0]?.text || '';
      const match = raw.match(/\[[\s\S]*\]/);
      const suggestions = match ? JSON.parse(match[0]).slice(0, 4) : [];
      res.json({ suggestions });
    } catch {
      res.json({ suggestions: [] });
    }
  });

  app.get('*', (req, res) => {
    res.sendFile(path.join(__dirname, 'out', 'index.html'));
  });

  const PORT = process.env.PORT || 3000;
  app.listen(PORT, () => console.log(`Rodando na porta ${PORT}`));
